\begin{algorithm}
    \floatname{algorithm}{Listing}
    \caption{Coordinator's reasoning process}
    \label{lst:coordinator}

    \begin{algorithmic}[1]
        \State {\textbf{Input:} \textbf{procedure} \immodFnlCall{assessStrategic}{0}}
        \State {\textbf{Input:} \textbf{procedure} \immodFnlCall{updateRequired}{0}}
        \Comment{Change in the global graph's upper level weights is required}
        \State {\textbf{Input:} \textbf{procedure} \immodFnlCall{checkGotTask}{0}}
        \Comment{A new task has arrived}
        \State {\textbf{Input:} node}
        \Comment{Any node from the cluster}

        \\

        \Procedure{run}{}
            \While{running}
                \If{\immodFnlCall{updateRequired}{}}
                    \State{ $W \gets $ \immodFnlCall{assessStrategic}{} }
                    \Comment{Update higher-level weights}
                \EndIf
                \If{task $\leftarrow$ \immodFnlCall{checkGotTask}{}}
                    \State{node.\immodFnlCall{onDecision}{task}}
                    \Comment{Pass the task}
                \EndIf
            \EndWhile
        \EndProcedure
    \end{algorithmic}

\end{algorithm}
